109 double visible inline __attribute__((always_inline)) carg(hr_complex_t c) {
110 double arg = 0;
111 if (creal(c) > 0) {
112 return atan(cimag(c) / creal(c));
113 } else if (creal(c) < 0 && cimag(c) >= 0) {
114 return atan(cimag(c) / creal(c)) + PI;
115 } else if (creal(c) < 0 && cimag(c) < 0) {
116 return atan(cimag(c) / creal(c)) - PI;
117 } else if (creal(c) == 0 && cimag(c) > 0) {
118 return PI / 2.0;
119 } else if (creal(c) == 0 && cimag(c) < 0) {
120 return -PI / 2.0;
121 } else if (creal(c) == 0 && creal(c) == 0) {
122 // This is technically undefined, so this will
123 // give 0 in the hopes that we get 0 by the
124 // mod and does not quit with error
125 return 0.0;
126 }
127 }

130 hr_complex_t visible inline __attribute__((always_inline)) cpow(hr_complex_t c, double pow) {
131 const double arg = carg(c);
132 const double mod = sqrt(creal(_complex_prod(c, c)));
133 return hr_complex_t(powf(mod, pow), 0) * hr_complex_t(cos(pow * arg), sin(pow * arg));
134 }
